\section{Quaternions as Rotations in ${R}^3$}

\subsection{Time Derivative}

\[
\dot{q} = \frac{1}{2} B \omega
\]

\[
\omega = 2 B^T \dot{q}
\]

\[
B
= \begin{bmatrix}
-b & -c & -d \\
 a & -d &  c \\
 d &  a & -b \\
-c &  b &  a
\end{bmatrix}
\]

\subsubsection{Time Derivative of Conjugate}

\[
\dot{q}^* = C \omega
\]

\[
C
= \begin{bmatrix}
-b & -c & -d \\
-a &  d & -c \\
-d & -a &  b \\
 c & -b & -a
\end{bmatrix}
\]
